// if yes then return true else false. // Input [10, 15, 3, 7], Target : 17 function addTwoNumbers(arr,target){ for (i = 0; i < arr.length ; i++){ for (j = 0; j < arr.length ; j++){ if (i !== j && (arr[i] + arr[j] === target){ return true; } } } return false } console.log(addTwoNumbers([10,15,3,7],18))
